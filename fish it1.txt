local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Fungsi Animasi Loading
local function StartLoading()
    local Window = Rayfield:CreateWindow({
       Name = "Joey Hub",
       LoadingTitle = "Joey Hub | Menyiapkan Sistem...",
       LoadingSubtitle = "Tunggu sebentar...",
       ConfigurationSaving = { Enabled = false },
       KeySystem = false,
    })

    -- Proses Loading 1% - 100%
    for i = 1, 100 do
        task.wait(0.03) -- Atur kecepatan loading di sini
        Window:ModifyLoadingSubtitle("Proses Memuat: " .. tostring(i) .. "%")
    end

    -- Mengganti Subtitle saat selesai
    Window:ModifyLoadingSubtitle("Berhasil Dimuat!")
    task.wait(0.5)
    
    return Window
end

-- Menjalankan Loading
local Window = StartLoading()

-- Setup Tema Pink
Rayfield:SetTheme("Pink")

-- Tab Utama (Sesuai permintaan: Hanya judul, Close, dan Minimize)
local MainTab = Window:CreateTab("Home", 4483362458) 
local Section = MainTab:CreateSection("Selamat Datang di Joey Hub")

-- Notifikasi Akhir
Rayfield:Notify({
   Title = "Selesai!",
   Content = "Joey Hub siap digunakan.",
   Duration = 3,
   Image = 4483362458,
})
