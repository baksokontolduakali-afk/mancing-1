local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- 1. Inisialisasi Window
local Window = Rayfield:CreateWindow({
   Name = "Joey Hub",
   LoadingTitle = "Joey Hub Executive",
   LoadingSubtitle = "Menyiapkan Data...",
   ConfigurationSaving = { Enabled = false },
   KeySystem = false,
})

-- 2. Animasi Loading 1% - 100%
for i = 1, 100 do
    task.wait(0.02)
    Window:ModifyLoadingSubtitle("Memuat: " .. i .. "%")
end

Window:ModifyLoadingSubtitle("Selesai!")
task.wait(0.3)

-- 3. Mengganti Tulisan Rayfield di UI menjadi Joey Hub & Set Warna Pink
Rayfield:SetTheme("Pink")

-- Fungsi ini akan mencari teks "Rayfield" di dalam GUI dan mengubahnya jadi "Joey Hub"
pcall(function()
    local coreGui = game:GetService("CoreGui")
    local ui = coreGui:FindFirstChild("Rayfield") or game:GetService("Players").LocalPlayer.PlayerGui:FindFirstChild("Rayfield")
    
    if ui then
        -- Mencari semua teks berlabel "Rayfield" di bagian bawah/samping
        for _, v in pairs(ui:GetDescendants()) do
            if v:IsA("TextLabel") and (v.Text:find("Rayfield") or v.Text:find("rayfield")) then
                v.Text = v.Text:gsub("Rayfield", "Joey Hub")
                v.Text = v.Text:gsub("rayfield", "Joey Hub")
            end
        end
    end
end)

-- 4. Membuat Konten (Hanya Tab & Section sesuai permintaan)
local Tab = Window:CreateTab("Home", 4483362458)
local Section = Tab:CreateSection("Joey Hub | Active")

-- Notifikasi Akhir
Rayfield:Notify({
   Title = "Joey Hub",
   Content = "Script Berhasil Terpasang!",
   Duration = 3,
})
