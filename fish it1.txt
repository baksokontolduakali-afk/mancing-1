ContentFrame.BackgroundTransparency = 1
ContentFrame.Parent = MainFrame

-- Fungsi Tween untuk Animasi
local function FadeIn(frame)
    frame.Visible = true
    frame.GroupTransparency = 1
    local tween = TweenService:Create(frame, TweenInfo.new(0.3), {GroupTransparency = 0})
    tween:Play()
end

-- Tab Management
local Tabs = {}
local function CreateTab(name)
    local TabButton = Instance.new("TextButton")
    TabButton.Size = UDim2.new(1, -10, 0, 35)
    TabButton.Position = UDim2.new(0, 5, 0, 45 + (#Sidebar:GetChildren() * 40))
    TabButton.BackgroundColor3 = Color3.fromRGB(30, 35, 40)
    TabButton.Text = name
    TabButton.TextColor3 = Color3.fromRGB(200, 200, 200)
    TabButton.Font = Enum.Font.Gotham
    TabButton.Parent = Sidebar
    
    local TabContent = Instance.new("CanvasGroup") -- CanvasGroup bagus untuk Fade Effect
    TabContent.Size = UDim2.new(1, 0, 1, 0)
    TabContent.BackgroundTransparency = 1
    TabContent.Visible = false
    TabContent.Parent = ContentFrame
    
    TabButton.MouseButton1Click:Connect(function()
        for _, v in pairs(ContentFrame:GetChildren()) do
            v.Visible = false
        end
        FadeIn(TabContent)
        
        -- Animasi warna tombol saat diklik
        TweenService:Create(TabButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(50, 150, 255)}):Play()
        task.wait(0.2)
        TweenService:Create(TabButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(30, 35, 40)}):Play()
    end)
    
    return TabContent
end
