local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Fungsi untuk memulai GUI dengan Loading
local function LaunchJoeyHub()
    local Window = Rayfield:CreateWindow({
       Name = "Joey Hub",
       LoadingTitle = "Joey Hub Executive",
       LoadingSubtitle = "Menyiapkan Data...",
       ConfigurationSaving = { Enabled = false },
       KeySystem = false,
    })

    -- Animasi Loading 1% - 100%
    for i = 1, 100 do
        task.wait(0.02) -- Kecepatan loading
        Window:ModifyLoadingSubtitle("Loading: " .. i .. "%")
    end

    Window:ModifyLoadingSubtitle("Selesai!")
    task.wait(0.5)

    -- PROSES MENGHAPUS TEKS/WATERMARK RAYFIELD
    pcall(function()
        local coreGui = game:GetService("CoreGui")
        local ui = coreGui:FindFirstChild("Rayfield") or game:GetService("Players").LocalPlayer.PlayerGui:FindFirstChild("Rayfield")
        if ui and ui:FindFirstChild("Main") then
            if ui.Main:FindFirstChild("Navigation") and ui.Main.Navigation:FindFirstChild("Badge") then
                ui.Main.Navigation.Badge:Destroy() -- Menghapus logo kecil
            end
            if ui.Main:FindFirstChild("Status") then
                ui.Main.Status:Destroy() -- Menghapus teks versi/watermark di bawah
            end
        end
    end)

    return Window
end

-- Eksekusi
local Window = LaunchJoeyHub()

-- Warna Tema Pink
Rayfield:SetTheme("Pink")

-- Tab Utama
local Tab = Window:CreateTab("Home", 4483362458)
local Section = Tab:CreateSection("Joey Hub | Active")

Rayfield:Notify({
   Title = "Joey Hub",
   Content = "Selamat datang!",
   Duration = 3,
})
